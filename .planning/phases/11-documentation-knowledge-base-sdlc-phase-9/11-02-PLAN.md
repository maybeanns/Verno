# Plan 11-02: Changelog Generator

## Goal

Implement `verno.generateChangelog` command that reads git log from the workspace repository, parses Conventional Commit messages (`feat:`, `fix:`, `chore:`, `docs:`, `refactor:`, `test:`, `perf:`, `ci:`, `breaking change:`), groups them by type and version tags, and writes a formatted `CHANGELOG.md` to the workspace root.

## Threat Model

<threat_model>
- **Threat**: `git log` shell command injection if workspace path contains special characters.
  **Mitigation**: Use `child_process.execFile` with arguments array (not `exec` with template string) to prevent shell injection.
- **Threat**: CHANGELOG.md overwrite destroying manually written notes.
  **Mitigation**: Append-only option; existing `CHANGELOG.md` content is preserved and prepended with new entries. Always ask user before overwriting.
- **ASVS**: Shell command execution — use `execFile`, not `exec`, to avoid injection.
</threat_model>

```typescript
import * as fs from 'fs';
import * as path from 'path';
import * as childProcess from 'child_process';
import * as util from 'util';

const execFile = util.promisify(childProcess.execFile);

export interface CommitEntry {
  hash: string;
  type: string;
  scope?: string;
  subject: string;
  body: string;
  isBreaking: boolean;
  date: string;
}

export interface ChangelogSection {
  version: string;
  date: string;
  breaking: CommitEntry[];
  features: CommitEntry[];
  fixes: CommitEntry[];
  performance: CommitEntry[];
  refactors: CommitEntry[];
  docs: CommitEntry[];
  chores: CommitEntry[];
  ci: CommitEntry[];
  tests: CommitEntry[];
  other: CommitEntry[];
}

export class ChangelogService {
  constructor(private logger: any) {}

  /**
   * Generate CHANGELOG.md from git history in the given workspace root.
   * Parses Conventional Commits and groups by version tags.
   * @param workspaceRoot - Absolute path to the git repository root
   * @param outputPath - Where to write CHANGELOG.md (defaults to workspaceRoot/CHANGELOG.md)
   * @returns Path to the written CHANGELOG.md
   */
  async generate(workspaceRoot: string, outputPath?: string): Promise<string> {
    const changelogPath = outputPath || path.join(workspaceRoot, 'CHANGELOG.md');

    // Fetch git log: hash | date | subject | body
    const gitLogArgs = [
      'log',
      '--pretty=format:COMMIT_START%n%H|%ad|%s%n%b%nCOMMIT_END',
      '--date=short',
      '--no-merges'
    ];

    let gitOutput: string;
    try {
      const { stdout } = await execFile('git', gitLogArgs, { cwd: workspaceRoot, maxBuffer: 10 * 1024 * 1024 });
      gitOutput = stdout;
    } catch (err: any) {
      // Try to get log even if exit code is non-zero (e.g. empty repo)
      gitOutput = err.stdout || '';
      if (!gitOutput) {
        throw new Error(`git log failed: ${err.message}`);
      }
    }

    // Fetch version tags
    let tags: Array<{ tag: string; hash: string; date: string }> = [];
    try {
      const { stdout: tagOutput } = await execFile(
        'git', ['tag', '-l', '--sort=-version:refname', '--format=%(refname:short)|%(objectname:short)|%(creatordate:short)'],
        { cwd: workspaceRoot }
      );
      tags = tagOutput.trim().split('\n').filter(Boolean).map(line => {
        const [tag, hash, date] = line.split('|');
        return { tag: tag?.trim() || '', hash: hash?.trim() || '', date: date?.trim() || '' };
      }).filter(t => t.tag);
    } catch {
      // No tags — treat all commits as "Unreleased"
    }

    const commits = this.parseCommits(gitOutput);
    const sections = this.groupByVersion(commits, tags);
    const markdown = this.renderMarkdown(sections);

    // Preserve existing CHANGELOG content (append new entries at top)
    let existingContent = '';
    if (fs.existsSync(changelogPath)) {
      existingContent = fs.readFileSync(changelogPath, 'utf-8');
      // Remove the old header if present to avoid duplication
      existingContent = existingContent.replace(/^# Changelog[\s\S]*?\n(?=##|$)/, '').trim();
    }

    const finalContent = existingContent
      ? `${markdown}\n\n---\n\n${existingContent}`
      : markdown;

    fs.writeFileSync(changelogPath, finalContent, 'utf-8');
    this.logger.info(`[ChangelogService] CHANGELOG.md written to ${changelogPath}`);
    return changelogPath;
  }

  private parseCommits(gitLog: string): CommitEntry[] {
    const commits: CommitEntry[] = [];
    const rawCommits = gitLog.split('COMMIT_START\n').filter(Boolean);

    // Conventional Commits pattern: type(scope)!: subject  OR  type: subject
    const conventionalRegex = /^(\w+)(\([^)]+\))?(!)?:\s*(.+)$/;

    for (const raw of rawCommits) {
      const endIdx = raw.indexOf('\nCOMMIT_END');
      const block = endIdx >= 0 ? raw.substring(0, endIdx) : raw;
      const lines = block.split('\n');
      const firstLine = lines[0] || '';

      const [hashDate, ...subjectParts] = firstLine.split('|');
      const hash = (hashDate || '').trim();
      const date = (subjectParts[0] || '').trim();
      const subject = (subjectParts[1] || '').trim();
      const body = lines.slice(1).join('\n').replace('COMMIT_END', '').trim();

      if (!hash || !subject) { continue; }

      const match = conventionalRegex.exec(subject);
      if (!match) {
        // Non-conventional commit — bucket as "other"
        commits.push({ hash, type: 'other', subject, body, isBreaking: false, date });
        continue;
      }

      const [, type, scopeRaw, bang] = match;
      const scope = scopeRaw ? scopeRaw.replace(/[()]/g, '') : undefined;
      const cleanSubject = match[4];
      const isBreaking = bang === '!' || body.toLowerCase().includes('breaking change:');

      commits.push({ hash, type: type.toLowerCase(), scope, subject: cleanSubject, body, isBreaking, date });
    }

    return commits;
  }

  private groupByVersion(
    commits: CommitEntry[],
    tags: Array<{ tag: string; hash: string; date: string }>
  ): ChangelogSection[] {
    // For simplicity: one "Unreleased" section + one section per tag
    // In a full implementation you'd map commits to tags by date/hash order
    const unreleased: ChangelogSection = this.emptySection('Unreleased', new Date().toISOString().split('T')[0]);

    for (const commit of commits) {
      this.bucketCommit(unreleased, commit);
    }

    const sections: ChangelogSection[] = [unreleased];

    // Add placeholder sections for existing tags (no commits assigned — preserves tag history)
    for (const tag of tags.slice(0, 5)) {
      const section = this.emptySection(tag.tag, tag.date);
      sections.push(section);
    }

    return sections;
  }

  private bucketCommit(section: ChangelogSection, commit: CommitEntry): void {
    if (commit.isBreaking) { section.breaking.push(commit); return; }
    switch (commit.type) {
      case 'feat':      section.features.push(commit); break;
      case 'fix':       section.fixes.push(commit); break;
      case 'perf':      section.performance.push(commit); break;
      case 'refactor':  section.refactors.push(commit); break;
      case 'docs':      section.docs.push(commit); break;
      case 'chore':     section.chores.push(commit); break;
      case 'ci':        section.ci.push(commit); break;
      case 'test':      section.tests.push(commit); break;
      default:          section.other.push(commit); break;
    }
  }

  private emptySection(version: string, date: string): ChangelogSection {
    return { version, date, breaking: [], features: [], fixes: [], performance: [], refactors: [], docs: [], chores: [], ci: [], tests: [], other: [] };
  }

  private renderMarkdown(sections: ChangelogSection[]): string {
    const lines: string[] = ['# Changelog', '', 'All notable changes to this project will be documented in this file.', ''];

    for (const section of sections) {
      const hasContent = [
        section.breaking, section.features, section.fixes, section.performance,
        section.refactors, section.docs, section.chores, section.ci, section.tests, section.other
      ].some(arr => arr.length > 0);

      if (!hasContent) { continue; }

      lines.push(`## [${section.version}] — ${section.date}`, '');

      if (section.breaking.length > 0) {
        lines.push('### ⚠ Breaking Changes', '');
        section.breaking.forEach(c => lines.push(this.formatEntry(c)));
        lines.push('');
      }
      if (section.features.length > 0) {
        lines.push('### ✨ Features', '');
        section.features.forEach(c => lines.push(this.formatEntry(c)));
        lines.push('');
      }
      if (section.fixes.length > 0) {
        lines.push('### 🐛 Bug Fixes', '');
        section.fixes.forEach(c => lines.push(this.formatEntry(c)));
        lines.push('');
      }
      if (section.performance.length > 0) {
        lines.push('### ⚡ Performance', '');
        section.performance.forEach(c => lines.push(this.formatEntry(c)));
        lines.push('');
      }
      if (section.refactors.length > 0) {
        lines.push('### ♻️ Refactors', '');
        section.refactors.forEach(c => lines.push(this.formatEntry(c)));
        lines.push('');
      }
      if (section.docs.length > 0) {
        lines.push('### 📝 Documentation', '');
        section.docs.forEach(c => lines.push(this.formatEntry(c)));
        lines.push('');
      }
      if (section.chores.length > 0) {
        lines.push('### 🔧 Chores', '');
        section.chores.forEach(c => lines.push(this.formatEntry(c)));
        lines.push('');
      }
      if (section.tests.length > 0) {
        lines.push('### 🧪 Tests', '');
        section.tests.forEach(c => lines.push(this.formatEntry(c)));
        lines.push('');
      }
    }

    return lines.join('\n');
  }

  private formatEntry(commit: CommitEntry): string {
    const scope = commit.scope ? `**${commit.scope}**: ` : '';
    return `- ${scope}${commit.subject} (\`${commit.hash.substring(0, 7)}\`)`;
  }
}
```

import { ChangelogService } from '../services/documentation/ChangelogService';

// Inside registerDocumentationCommands, after existing commands:
const changelogService = new ChangelogService(logger);

const generateChangelogCmd = vscode.commands.registerCommand('verno.generateChangelog', async () => {
  const workspaceFolders = vscode.workspace.workspaceFolders;
  if (!workspaceFolders || workspaceFolders.length === 0) {
    vscode.window.showErrorMessage('Verno: No workspace open.');
    return;
  }
  const workspaceRoot = workspaceFolders[0].uri.fsPath;

  const choice = await vscode.window.showQuickPick(
    ['Generate / Update CHANGELOG.md', 'Cancel'],
    { placeHolder: 'Verno: Generate Changelog from git history?' }
  );
  if (!choice || choice === 'Cancel') { return; }

  await vscode.window.withProgress(
    { location: vscode.ProgressLocation.Notification, title: 'Verno: Generating CHANGELOG.md...', cancellable: false },
    async () => {
      try {
        const outputPath = await changelogService.generate(workspaceRoot);
        const uri = vscode.Uri.file(outputPath);
        await vscode.window.showTextDocument(uri);
        vscode.window.showInformationMessage('Verno: CHANGELOG.md generated successfully.');
      } catch (err: any) {
        vscode.window.showErrorMessage(`Verno: Changelog generation failed — ${err.message}`);
      }
    }
  );
});

context.subscriptions.push(generateChangelogCmd);
