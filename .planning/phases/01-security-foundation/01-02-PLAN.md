## Step 1 — Create validateWebviewMessage utility

Create `src/utils/webviewSecurity.ts`:

```typescript
export interface WebviewMessage {
    type: string;
    [key: string]: unknown;
}

export function validateWebviewMessage<T extends WebviewMessage>(
    message: unknown,
    allowedTypes: readonly string[],
    logger?: { warn: (msg: string) => void }
): message is T {
    if (!message || typeof message !== 'object') {
        logger?.warn(`[WebviewSecurity] Invalid message format: ${JSON.stringify(message)}`);
        return false;
    }
    const msg = message as Record<string, unknown>;
    if (typeof msg.type !== 'string') {
        logger?.warn(`[WebviewSecurity] Missing or invalid message.type`);
        return false;
    }
    if (!allowedTypes.includes(msg.type)) {
        logger?.warn(`[WebviewSecurity] Unknown message type: "${msg.type}". Allowed: ${allowedTypes.join(', ')}`);
        return false;
    }
    return true;
}
```

## Step 2 — Update AgentPanel

In `src/ui/panels/AgentPanel.ts`, define allowlist:

```typescript
const ALLOWED_MESSAGE_TYPES = [
    'processInput', 'listConversations', 'loadConversation', 
    'deleteConversation', 'newConversation', 'newTask',
    'startRecording', 'stopRecording', 'startSDLC', 'mcpInstall'
] as const;
```

Wrap the `onDidReceiveMessage` handler:
```typescript
if (!validateWebviewMessage(message, ALLOWED_MESSAGE_TYPES, this.logger)) return;
```

## Step 5 — CSP Audit

For each webview HTML template, verify CSP includes:
- `default-src 'none'`
- `script-src 'nonce-${nonce}'`
- `style-src 'nonce-${nonce}' 'unsafe-inline'` (allow inline styles only)
- `img-src ${webview.cspSource} data:`
- `font-src ${webview.cspSource}`

Remove any `script-src 'unsafe-inline'` entries.
